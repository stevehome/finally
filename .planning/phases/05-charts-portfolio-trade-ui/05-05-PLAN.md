              Watchlist
            </div>
            <WatchlistPanel
              tickers={tickers}
              prices={prices}
              sparkHistory={sparkHistory}
              selectedTicker={selectedTicker}
              onSelectTicker={setSelectedTicker}
            />
          </div>

          {/* Main area — nested grid */}
          <div style={{
            display: 'grid',
            gridTemplateRows: '60% 40%',
            height: '100%',
            overflow: 'hidden',
            backgroundColor: '#0d1117',

              ticker={selectedTicker}
              sparkHistory={sparkHistory}
              prices={prices}
            />

              display: 'grid',

              overflow: 'hidden',
            }}>
              {/* Portfolio panels + trade bar */}
              <div style={{
                borderRight: '1px solid #30363d',
                overflow: 'hidden',
                display: 'flex',
                flexDirection: 'column',
              }}>
                <TradeBar refetch={refetch} />
                <div style={{ flex: 1, overflow: 'hidden' }}>
                  <PortfolioPanels portfolio={portfolio} />
                </div>
              </div>

              {/* Chat panel */}
              <ChatPanel />
            </div>
          </div>
        </div>
      )
    }
